# How to install Virtual_Environment in Ubuntu 20.04

$sudo apt update

$sudo apt install python3.8

$sudo apt install python3-venv

 $python3 -m venv ENV
 
 ### Activate
 $source env/bin/activate
 
 ### Deactivate
 $deactivate


$ export FLASK_APP=hello.py

$ export FLASK_ENV=development
